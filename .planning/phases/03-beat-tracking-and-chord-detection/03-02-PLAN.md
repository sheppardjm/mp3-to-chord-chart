    ```python
    # Chromatic note names for chord labeling.
    # Flat-side convention matching key_detection.py ROOTS.
    NOTES = ['C', 'Db', 'D', 'Eb', 'E', 'F', 'F#', 'G', 'Ab', 'A', 'Bb', 'B']


    def build_chord_templates() -> tuple[list[str], np.ndarray]:
        """
        Build normalized chord template matrix for cosine similarity matching.

        Creates 36 templates: 12 roots x 3 qualities (major, minor, dominant 7th).
        Each template is a 12-dimensional unit vector representing the expected
        chroma energy distribution for that chord.

        Returns:
            (chord_names, template_matrix) where:
                - chord_names: list of 36 strings like "C:maj", "C:min", "C:7"
                - template_matrix: np.ndarray of shape (36, 12), L2-normalized rows
        """
        chord_names = []
        rows = []

        for root in range(12):
            for suffix, intervals in [(':maj', [0, 4, 7]),
                                       (':min', [0, 3, 7]),
                                       (':7', [0, 4, 7, 10])]:
                t = np.zeros(12)
                for i in intervals:
                    t[(root + i) % 12] = 1.0
                t /= np.linalg.norm(t)  # L2 normalize for cosine similarity
                chord_names.append(f"{NOTES[root]}{suffix}")
                rows.append(t)

        return chord_names, np.array(rows)  # (36, 12)


    def detect_chords(chroma_sync: np.ndarray, chord_names: list[str],
                      template_matrix: np.ndarray) -> list[str]:
        """
        Detect chords via cosine similarity against templates + Viterbi smoothing.

        For each beat, computes cosine similarity between the beat's chroma vector
        and all 36 chord templates. The resulting probability matrix is smoothed
        with Viterbi decoding using a self-loop transition matrix (p_loop=0.5).

        Args:
            chroma_sync: (12, n_beats) beat-synchronized chroma from extract_beat_chroma().
            chord_names: List of 36 chord name strings from build_chord_templates().
            template_matrix: (36, 12) normalized template matrix from build_chord_templates().

        Returns:
            List of chord name strings, one per beat. Length matches chroma_sync columns.

        NOTE: p_loop=0.5 is calibrated for pop/rock songs. At 0.8+ the Viterbi
        over-smooths (entire song = 1 chord). At 0.3 or below, transitions are
        too frequent (noisy output). 0.5 gives ~5 distinct segments for a
        3-4 minute song (validated on Don't Cave.mp3).
        """
        # Normalize chroma per-frame for cosine similarity
        chroma_norm = chroma_sync / (np.linalg.norm(chroma_sync, axis=0, keepdims=True) + 1e-8)

        # Cosine similarity: (36, 12) @ (12, n_beats) = (36, n_beats)
        probs = template_matrix @ chroma_norm
        probs = np.clip(probs, 0, None)  # cosine sim can be negative; floor at 0
        probs = probs / (probs.sum(axis=0, keepdims=True) + 1e-8)  # normalize to probabilities

        # Viterbi smoothing — transition_loop creates (36, 36) matrix with
        # p_loop on diagonal and (1-p_loop)/35 on off-diagonals.
        # sequence._viterbi uses @jit (not @guvectorize) — works with numba stub.
        transition = librosa.sequence.transition_loop(len(chord_names), 0.5)
        states = librosa.sequence.viterbi(probs, transition)

        return [chord_names[s] for s in states]


    def collapse_chords(chord_seq: list[str], beat_times: np.ndarray) -> list[dict]:
        """
        Collapse consecutive identical chord labels into segments with timestamps.

        A 4-bar C major chord (16 beats) becomes one entry: {"chord": "C:maj", "time": 2.3}.
        This produces the human-readable chord timeline needed by downstream phases.

        Args:
            chord_seq: List of chord names, one per beat (from detect_chords()).
            beat_times: np.ndarray of timestamps in seconds for each beat.
                        Must have same length as chord_seq (or be truncated to match).

        Returns:
            List of dicts with keys:
                - "chord": str — chord name (e.g. "G:maj", "C:7")
                - "time": float — timestamp in seconds where this chord starts
        """
        result = []
        for chord, time in zip(chord_seq, beat_times):
            if not result or result[-1]['chord'] != chord:
                result.append({'chord': chord, 'time': float(time)})
        return result


    def detect_chords_pipeline(audio: dict) -> dict:
        """
        Full chord detection pipeline: beat tracking -> chroma sync -> templates -> Viterbi -> collapse.

        This is the main entry point for chord detection. Takes the output of
        load_audio() and returns a complete chord analysis.

        Args:
            audio: dict from loader.load_audio() with keys:
                   y, sr, y_harmonic, y_percussive, duration

        Returns:
            dict with keys:
                - tempo_bpm: float — estimated tempo
                - beat_times: list[float] — timestamp for each beat (seconds)
                - chord_sequence: list[str] — chord label per beat
                - chord_segments: list[dict] — collapsed {chord, time} segments
                - n_beats: int — total number of beats detected
                - n_segments: int — number of unique chord segments after collapse
        """
        # 1. Beat tracking (grid-based, numba workaround)
        tempo_bpm, beat_frames = beat_track_grid(audio['y_percussive'], audio['sr'])

        # 2. Beat-synced chroma
        chroma_sync = extract_beat_chroma(audio['y_harmonic'], beat_frames, audio['sr'])

        # 3. Chord templates
        chord_names, template_matrix = build_chord_templates()

        # 4. Chord detection with Viterbi
        chord_seq = detect_chords(chroma_sync, chord_names, template_matrix)

        # 5. Beat timestamps (truncated to match chord_seq length)
        beat_frames_fixed = librosa.util.fix_frames(beat_frames, x_min=0)
        beat_times = librosa.frames_to_time(
            beat_frames_fixed[:len(chord_seq)], sr=audio['sr'], hop_length=HOP
        )

        # 6. Collapse consecutive identical chords
        segments = collapse_chords(chord_seq, beat_times)

        return {
            'tempo_bpm': tempo_bpm,
            'beat_times': beat_times.tolist(),
            'chord_sequence': chord_seq,
            'chord_segments': segments,
            'n_beats': len(chord_seq),
            'n_segments': len(segments),
        }
    ```

    Implementation notes:
    - NOTES list matches ROOTS in key_detection.py (flat-side convention)
    - build_chord_templates() creates exactly 36 templates: 12 roots x 3 qualities
    - Template order is root-grouped: C:maj, C:min, C:7, Db:maj, Db:min, Db:7, ...
    - detect_chords() clips negative cosine similarities to 0 before normalizing
    - transition_loop(36, 0.5) gives 50% self-loop probability — researched empirically
    - collapse_chords() is a simple itertools-style collapse with timestamp tracking
    - detect_chords_pipeline() uses fix_frames on beat_frames for frames_to_time alignment
    - Pipeline returns both per-beat sequence AND collapsed segments

    Do NOT:
    - Use p_loop > 0.7 — over-smooths to 1-2 chords (Pitfall 3 in research)
    - Normalize chroma globally — must be per-frame (axis=0) for cosine similarity
    - Include chord qualities beyond :maj, :min, :7 — out of scope for v1
    - Use argmax without Viterbi — produces 42% change rate (unusable noise)
    - Call beat_track() or any numba @guvectorize functions
