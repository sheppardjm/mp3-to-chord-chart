    def measure_accuracy(detected: list[str], reference: list[str]) -> dict:
        """
        Measure chord detection accuracy at beat level.

        Compares detected chord labels against reference labels, both as exact
        match and root-only match (ignoring quality — :maj/:min/:7).

        Both lists must be the same length (aligned by beat index). If lengths
        differ, comparison is truncated to the shorter list.

        Args:
            detected: List of detected chord strings (e.g. ["G:maj", "C:maj", ...])
            reference: List of reference chord strings (same format)

        Returns:
            dict with:
                - exact_accuracy: float — fraction of beats where chord matches exactly
                - root_accuracy: float — fraction of beats where root note matches
                - n_compared: int — number of beats compared
                - exact_matches: int — count of exact matches
                - root_matches: int — count of root-only matches
        """
        n = min(len(detected), len(reference))
        if n == 0:
            return {'exact_accuracy': 0.0, 'root_accuracy': 0.0,
                    'n_compared': 0, 'exact_matches': 0, 'root_matches': 0}

        exact_matches = 0
        root_matches = 0
        for d, r in zip(detected[:n], reference[:n]):
            if d == r:
                exact_matches += 1
            d_root = d.split(':')[0]
            r_root = r.split(':')[0]
            if d_root == r_root:
                root_matches += 1

        return {
            'exact_accuracy': exact_matches / n,
            'root_accuracy': root_matches / n,
            'n_compared': n,
            'exact_matches': exact_matches,
            'root_matches': root_matches,
        }
    ```

    """Structural tests for chord detection pipeline.

    These tests validate invariants of the pipeline functions using synthetic data.
    They do NOT require audio files and can run in CI.
    """
    import numpy as np
    import sys
    import os

    # Add backend to path for imports
    sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..'))

    from audio.chord_detection import (
        HOP, NOTES, build_chord_templates, detect_chords,
        collapse_chords, measure_accuracy,
    )


    def test_hop_constant():
        """HOP must be 1024 to match Phase 2 HPSS."""
        assert HOP == 1024


    def test_notes_list():
        """NOTES must have 12 entries matching key_detection.py ROOTS."""
        assert len(NOTES) == 12
        assert NOTES[0] == 'C'
        assert NOTES[7] == 'G'
        assert NOTES[6] == 'F#'
        assert NOTES[10] == 'Bb'


    def test_build_chord_templates_count():
        """Must produce exactly 36 templates (12 roots x 3 qualities)."""
        names, matrix = build_chord_templates()
        assert len(names) == 36
        assert matrix.shape == (36, 12)


    def test_build_chord_templates_normalized():
        """All template rows must be L2-normalized (unit vectors)."""
        _, matrix = build_chord_templates()
        norms = np.linalg.norm(matrix, axis=1)
        assert np.allclose(norms, 1.0, atol=1e-6)


    def test_build_chord_templates_names():
        """Template names must follow ROOT:quality format with valid suffixes."""
        names, _ = build_chord_templates()
        valid_suffixes = {':maj', ':min', ':7'}
        for name in names:
            assert ':' in name, f"Missing colon in chord name: {name}"
            suffix = name[name.index(':'):]
            assert suffix in valid_suffixes, f"Invalid suffix in: {name}"


    def test_build_chord_templates_all_roots():
        """All 12 roots must appear in template names."""
        names, _ = build_chord_templates()
        roots = set(name.split(':')[0] for name in names)
        assert roots == set(NOTES)


    def test_detect_chords_output_length():
        """detect_chords must return one chord per beat (column)."""
        names, matrix = build_chord_templates()
        n_beats = 50
        # Synthetic chroma: random non-negative values
        chroma = np.random.rand(12, n_beats) + 0.1
        chords = detect_chords(chroma, names, matrix)
        assert len(chords) == n_beats


    def test_detect_chords_valid_names():
        """All returned chords must be from the template name list."""
        names, matrix = build_chord_templates()
        chroma = np.random.rand(12, 30) + 0.1
        chords = detect_chords(chroma, names, matrix)
        name_set = set(names)
        for chord in chords:
            assert chord in name_set, f"Unknown chord: {chord}"


    def test_detect_chords_pure_c_major():
        """A pure C major chroma pattern should detect as C:maj."""
        names, matrix = build_chord_templates()
        # C major = pitch classes 0 (C), 4 (E), 7 (G)
        chroma = np.zeros((12, 20))
        chroma[0, :] = 1.0  # C
        chroma[4, :] = 1.0  # E
        chroma[7, :] = 1.0  # G
        chords = detect_chords(chroma, names, matrix)
        # With Viterbi smoothing, should converge on C:maj
        from collections import Counter
        counts = Counter(chords)
        most_common = counts.most_common(1)[0][0]
        assert most_common == 'C:maj', f"Expected C:maj, got {most_common}"


    def test_detect_chords_pure_g_major():
        """A pure G major chroma pattern should detect as G:maj."""
        names, matrix = build_chord_templates()
        chroma = np.zeros((12, 20))
        chroma[7, :] = 1.0   # G
        chroma[11, :] = 1.0  # B
        chroma[2, :] = 1.0   # D
        chords = detect_chords(chroma, names, matrix)
        from collections import Counter
        counts = Counter(chords)
        most_common = counts.most_common(1)[0][0]
        assert most_common == 'G:maj', f"Expected G:maj, got {most_common}"


    def test_collapse_no_duplicates():
        """Collapsed output must not have adjacent identical chords."""
        chords = ['C:maj', 'C:maj', 'G:maj', 'G:maj', 'G:maj', 'C:maj']
        times = np.array([0.0, 0.5, 1.0, 1.5, 2.0, 2.5])
        segments = collapse_chords(chords, times)
        for i in range(1, len(segments)):
            assert segments[i]['chord'] != segments[i-1]['chord']


    def test_collapse_preserves_first_timestamp():
        """Each segment's timestamp should be the first beat of that chord."""
        chords = ['C:maj', 'C:maj', 'G:maj', 'G:maj']
        times = np.array([0.0, 0.5, 1.0, 1.5])
        segments = collapse_chords(chords, times)
        assert len(segments) == 2
        assert segments[0] == {'chord': 'C:maj', 'time': 0.0}
        assert segments[1] == {'chord': 'G:maj', 'time': 1.0}


    def test_collapse_single_chord():
        """A single repeated chord collapses to one segment."""
        chords = ['G:maj'] * 100
        times = np.arange(100) * 0.5
        segments = collapse_chords(chords, times)
        assert len(segments) == 1
        assert segments[0]['chord'] == 'G:maj'
        assert segments[0]['time'] == 0.0


    def test_collapse_all_different():
        """All different chords produce one segment per beat."""
        chords = ['C:maj', 'D:min', 'E:min', 'F:maj']
        times = np.array([0.0, 1.0, 2.0, 3.0])
        segments = collapse_chords(chords, times)
        assert len(segments) == 4


    def test_measure_accuracy_perfect():
        """Perfect match should return 1.0 accuracy."""
        detected = ['C:maj', 'G:maj', 'D:min']
        reference = ['C:maj', 'G:maj', 'D:min']
        result = measure_accuracy(detected, reference)
        assert result['exact_accuracy'] == 1.0
        assert result['root_accuracy'] == 1.0
        assert result['n_compared'] == 3


    def test_measure_accuracy_root_match():
        """Root match with different quality should count for root_accuracy only."""
        detected = ['C:maj', 'G:7', 'D:min']
        reference = ['C:min', 'G:maj', 'D:maj']
        result = measure_accuracy(detected, reference)
        assert result['exact_accuracy'] == 0.0  # none match exactly
        assert result['root_accuracy'] == 1.0  # all roots match


    def test_measure_accuracy_empty():
        """Empty lists should return 0.0 accuracy."""
        result = measure_accuracy([], [])
        assert result['exact_accuracy'] == 0.0
        assert result['n_compared'] == 0


    def test_measure_accuracy_length_mismatch():
        """Mismatched lengths should compare up to the shorter list."""
        detected = ['C:maj', 'G:maj', 'D:min', 'A:min']
        reference = ['C:maj', 'G:maj']
        result = measure_accuracy(detected, reference)
        assert result['n_compared'] == 2
        assert result['exact_accuracy'] == 1.0


    def run_all_tests():
        """Run all tests and report results."""
        import inspect
        tests = [(name, func) for name, func in globals().items()
                 if name.startswith('test_') and callable(func)]

        passed = 0
        failed = 0
        for name, func in sorted(tests):
            try:
                func()
                print(f"  PASS: {name}")
                passed += 1
            except Exception as e:
                print(f"  FAIL: {name}: {e}")
                failed += 1

        print(f"\n{passed} passed, {failed} failed, {passed + failed} total")
        if failed > 0:
            raise SystemExit(1)
        return passed, failed


    if __name__ == '__main__':
        run_all_tests()
    ```
